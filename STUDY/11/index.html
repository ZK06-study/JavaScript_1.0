JavaScript 배열 완전 정복

순서가 있는 데이터를 다루는 가장 강력한 도구, 배열(Array)의 모든 것

배열이란?

배열은 순서가 있는 값들의 집합입니다. 각 값은 고유한 번호(인덱스)로 접근할 수 있습니다.

// 배열 생성하기
const numbers = [10, 20, 30, 40, 50];
const fruits = ['사과', '바나나', '딸기'];
const mixed = [42, 'hello', true, { name: 'Kim' }];

// 배열의 특징
console.log(typeof numbers);  // 'object' (배열은 특별한 객체)
console.log(numbers.length);    // 5 (요소의 개수)
console.log(Array.isArray(numbers)); // true

핵심 특징

  • • 인덱스는 0부터 시작
  • • length 속성 자동 관리
  • • 다양한 타입 저장 가능
  • • 동적 크기 조절

배열 생성 방법

// 1. 배열 리터럴 (권장)
const arr1 = [1, 2, 3];

// 2. Array 생성자
const arr2 = new Array(5);        // 길이가 5인 빈 배열
const arr3 = new Array(1, 2, 3); // [1, 2, 3]

// 3. Array.of() - ES6
const arr4 = Array.of(7);         // [7] (숫자 7을 요소로)

// 4. Array.from() - 유사 배열을 배열로 변환
const str = 'hello';
const chars = Array.from(str);    // ['h', 'e', 'l', 'l', 'o']

// 5. 특수한 배열 생성
const zeros = new Array(5).fill(0); // [0, 0, 0, 0, 0]
const range = Array.from({length: 5}, (_, i) => i + 1); // [1, 2, 3, 4, 5]

배열 요소 접근하기

배열의 각 요소는 인덱스를 통해 접근합니다. 클릭해보세요!

요소를 클릭하면 결과가 표시됩니다
const fruits = ['🍎', '🍌', '🍓', '🍊', '🍇'];

// 인덱스로 접근
console.log(fruits[0]);  // '🍎' (첫 번째 요소)
console.log(fruits[4]);  // '🍇' (마지막 요소)
console.log(fruits[10]); // undefined (범위 초과)

// 음수 인덱스는 at() 메소드로
console.log(fruits.at(-1)); // '🍇' (마지막 요소)

끝에서 추가/제거: push & pop

결과가 여기에 표시됩니다
const stack = ['🍎', '🍌'];

// push: 끝에 추가 (반환값: 새 길이)
let newLength = stack.push('🍓');
console.log(stack);      // ['🍎', '🍌', '🍓']
console.log(newLength);  // 3

// pop: 끝에서 제거 (반환값: 제거된 요소)
let removed = stack.pop();
console.log(removed);    // '🍓'
console.log(stack);      // ['🍎', '🍌']

앞에서 추가/제거: unshift & shift

결과가 여기에 표시됩니다
const queue = ['🍎', '🍌'];

// unshift: 앞에 추가 (반환값: 새 길이)
let newLength = queue.unshift('🥝');
console.log(queue);      // ['🥝', '🍎', '🍌']
console.log(newLength);  // 3

// shift: 앞에서 제거 (반환값: 제거된 요소)
let first = queue.shift();
console.log(first);      // '🥝'
console.log(queue);      // ['🍎', '🍌']

// 성능 팁: push/pop이 unshift/shift보다 빠릅니다!

중간 조작: splice

splice()는 배열의 어느 위치에서든 요소를 추가, 제거, 교체할 수 있는 만능 메소드입니다.

let arr = ['a', 'b', 'c', 'd', 'e'];

// splice(시작인덱스, 제거개수, 추가할요소들...)

// 1. 요소 제거
let removed = arr.splice(2, 2);  // 인덱스 2부터 2개 제거
console.log(removed);  // ['c', 'd']
console.log(arr);      // ['a', 'b', 'e']

// 2. 요소 추가
arr.splice(2, 0, 'X', 'Y');  // 인덱스 2에 추가
console.log(arr);      // ['a', 'b', 'X', 'Y', 'e']

// 3. 요소 교체
arr.splice(1, 2, '🌟');     // 1번 인덱스부터 2개를 '🌟'로 교체
console.log(arr);      // ['a', '🌟', 'Y', 'e']

변환하기: map()

map()은 각 요소를 변환하여 새로운 배열을 만듭니다.

const prices = [1000, 2000, 3000];

// 10% 할인 적용
const discounted = prices.map(price => {
  return price * 0.9;
});

// 화폐 형식으로 변환
const formatted = prices.map(price => {
  return `₩${price.toLocaleString()}`;
});

// 인덱스도 활용 가능
const indexed = prices.map((price, index) => {
  return `${index + 1}번: ${price}원`;
});

실행 결과

discounted: [900, 1800, 2700]

formatted: ['₩1,000', '₩2,000', '₩3,000']

indexed: ['1번: 1000원', '2번: 2000원', '3번: 3000원']

걸러내기: filter()

filter()는 조건을 만족하는 요소만 모아 새로운 배열을 만듭니다.

const students = [
  { name: '김철수', score: 85, passed: true },
  { name: '이영희', score: 92, passed: true },
  { name: '박민수', score: 68, passed: false },
  { name: '최지원', score: 78, passed: true }
];

// 합격자만 필터링
const passed = students.filter(student => {
  return student.passed;
});

// 80점 이상 우수 학생
const excellent = students.filter(student => {
  return student.score >= 80;
});

// 복합 조건
const excellentPassed = students.filter(s => {
  return s.score >= 80 && s.passed;
});

필터링 결과

passed: 3명 (김철수, 이영희, 최지원)
excellent: 2명 (김철수, 이영희)
excellentPassed: 2명

집계하기: reduce()

reduce()는 배열을 하나의 값으로 축약합니다. 가장 강력한 메소드!

const numbers = [1, 2, 3, 4, 5];

// 기본 형태: reduce((누적값, 현재값) => 다음누적값, 초기값)

// 1. 합계 구하기
const sum = numbers.reduce((acc, num) => acc + num, 0);
console.log(sum); // 15

// 2. 최댓값 찾기
const max = numbers.reduce((acc, num) => Math.max(acc, num), -Infinity);

// 3. 객체로 그룹핑
const words = ['apple', 'banana', 'apple', 'cherry', 'banana'];
const count = words.reduce((acc, word) => {
  acc[word] = (acc[word] || 0) + 1;
  return acc;
}, {});
// { apple: 2, banana: 2, cherry: 1 }

// 4. 파이프라인 처리
const pipeline = [
  x => x * 2,
  x => x + 10,
  x => x / 5
];
const result = pipeline.reduce((value, fn) => fn(value), 10);
// 10 → 20 → 30 → 6

찾기: find & findIndex

조건에 맞는 첫 번째 요소를 찾습니다.

const users = [
  { id: 1, name: 'Alice', age: 25 },
  { id: 2, name: 'Bob', age: 30 },
  { id: 3, name: 'Charlie', age: 25 }
];

// find: 요소 자체를 반환
const user = users.find(u => u.id === 2);
console.log(user); // { id: 2, name: 'Bob', age: 30 }

// findIndex: 인덱스를 반환
const index = users.findIndex(u => u.age === 25);
console.log(index); // 0 (첫 번째 일치)

// 못 찾으면?
const notFound = users.find(u => u.age > 50);
console.log(notFound); // undefined

const notFoundIndex = users.findIndex(u => u.age > 50);
console.log(notFoundIndex); // -1

다른 검색 메소드들

  • • includes() - 포함 여부 (true/false)
  • • indexOf() - 값의 인덱스
  • • some() - 하나라도 만족?
  • • every() - 모두 만족?

메소드 체이닝

여러 메소드를 연결하여 강력한 데이터 파이프라인을 만들 수 있습니다.

const products = [
  { name: '노트북', price: 1200000, category: '전자제품', stock: 5 },
  { name: '마우스', price: 25000, category: '액세서리', stock: 0 },
  { name: '키보드', price: 75000, category: '액세서리', stock: 10 },
  { name: '모니터', price: 300000, category: '전자제품', stock: 3 }
];

// 재고가 있는 전자제품 중 가격이 높은 순으로 정렬하여 이름만 추출
const result = products
  .filter(p => p.stock > 0)                    // 재고 있는 것만
  .filter(p => p.category === '전자제품')      // 전자제품만
  .sort((a, b) => b.price - a.price)           // 가격 내림차순
  .map(p => ({                                 // 필요한 정보만
    name: p.name,
    displayPrice: `₩${p.price.toLocaleString()}`
  }));

console.log(result);
// [
//   { name: '노트북', displayPrice: '₩1,200,000' },
//   { name: '모니터', displayPrice: '₩300,000' }
// ]

// 총 재고 가치 계산
const totalValue = products
  .filter(p => p.stock > 0)
  .map(p => p.price * p.stock)
  .reduce((sum, value) => sum + value, 0);

console.log(`총 재고 가치: ₩${totalValue.toLocaleString()}`);
// 총 재고 가치: ₩7,650,000

실무 활용 팁

// 1. 구조 분해 할당 활용
const [first, second, ...rest] = [1, 2, 3, 4, 5];
console.log(first);  // 1
console.log(rest);   // [3, 4, 5]

// 2. 배열 복사 (얕은 복사)
const original = [1, 2, 3];
const copy1 = [...original];        // 스프레드 연산자
const copy2 = original.slice();     // slice 메소드
const copy3 = Array.from(original); // Array.from

// 3. 배열 합치기
const arr1 = [1, 2];
const arr2 = [3, 4];
const combined = [...arr1, ...arr2]; // [1, 2, 3, 4]

// 4. 중복 제거
const numbers = [1, 2, 2, 3, 3, 4];
const unique = [...new Set(numbers)]; // [1, 2, 3, 4]

// 5. 배열 평탄화
const nested = [[1, 2], [3, [4, 5]]];
const flat1 = nested.flat();      // [1, 2, 3, [4, 5]]
const flat2 = nested.flat(2);     // [1, 2, 3, 4, 5]
const flatInf = nested.flat(Infinity); // 모든 중첩 해제
