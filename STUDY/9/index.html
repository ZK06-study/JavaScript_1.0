제어 구문

Control Statements

조건문과 반복문으로 프로그램의 논리 구조 만들기

학습 목표

  • 조건문(if, switch)을 이용한 분기 처리
  • 반복문(while, for)을 이용한 반복 처리
  • break, continue를 이용한 실행 흐름 제어

제어 구문이란?

프로그램의 논리 구조

실제 프로그램은 단순한 계산이 아닌 복잡한 논리 구조를 가지고 있습니다.

경우의 수 처리

다양한 상황과 조건에 따라 다른 동작을 수행해야 합니다.

실생활 예시

클릭 이벤트: 왼쪽/오른쪽/더블클릭에 따른 다른 동작

게임: 아이템 종류에 따른 효과 적용

폼 검증: 입력값 유효성에 따른 처리

조건문 (Conditional Statement)

조건문의 역할

특정 조건을 만족할 때와 그렇지 않을 때 다른 코드를 실행하도록 합니다.

기본 형태

if (조건) {
  // 조건이 참일 때 실행
} else {
  // 조건이 거짓일 때 실행
}

if...else 구문

주사위 게임 예제

function roll() {
  return Math.ceil(Math.random() * 6);
}

function game() {
  const result = roll();
  alert(`결과: ${result}`);

  // if...else 구문
  if (result >= 4) {
    // 조건이 true이면 실행
    alert('당신이 이겼습니다!');
  } else {
    // 조건이 false이면 실행
    alert('당신이 졌습니다.');
  }
}

game();

if...else 변형

else 생략

if (result === 6) {
  alert('당신은 운이 좋군요!');
}

중괄호 생략

if (result === 6) alert('당신은 운이 좋군요!');
한 줄일 때만 가능

권장 사항

  • 가독성을 위해 중괄호 사용 권장
  • 조건이 복잡할 때는 여러 줄로 작성
  • 일관된 스타일 유지

다중 조건문

중첩된 형태

if (result >= 5) {
  alert('당신이 이겼습니다!');
} else {
  if (result >= 3) {
    alert('비겼습니다.');
  } else {
    alert('당신이 졌습니다.');
  }
}

개선된 형태

if (result >= 5) {
  alert('당신이 이겼습니다!');
} else if (result >= 3) {
  alert('비겼습니다.');
} else {
  alert('당신이 졌습니다.');
}

switch 구문

언제 사용하나?

하나의 변수에 대해 많은 경우의 수가 있을 때

if...else 버전

function translateColor(english) {
  if (english === 'red') {
    return '빨강색';
  } else if (english === 'blue') {
    return '파랑색';
  } else if (english === 'purple' || english === 'violet') {
    return '보라색';
  } else {
    return '일치하는 색깔이 없습니다.';
  }
}

switch 버전

function translateColor(english) {
  let result;
  switch (english) {
    case 'red':
      result = '빨강색';
      break;
    case 'blue':
      result = '파랑색';
      break;
    case 'purple':
    case 'violet':
      result = '보라색';
      break;
    default:
      result = '일치하는 색깔이 없습니다.';
  }
  return result;
}

break의 중요성

주의: break를 빼먹으면?

break 없음 (문제)

switch (english) {
  case 'blue':
    result = '파랑색';
    // break가 없음!
  case 'purple':
    result = '보라색';
    break;
}

translateColor('blue'); // '보라색' (잘못된 결과)

break 있음 (정상)

switch (english) {
  case 'blue':
                            result = '파랑색';
    break; // 중요!
  case 'purple':
    result = '보라색';
    break;
}

translateColor('blue'); // '파랑색' (올바른 결과)

반복문 (Looping Statement)

왜 반복문이 필요한가?

스프레드시트: 1000개 행의 데이터 처리

게임: 승부가 날 때까지 턴 반복

댓글: 모든 댓글을 화면에 표시

반복문의 종류

while

조건이 참인 동안 반복

do...while

최소 한 번은 실행

for

횟수가 정해진 반복

while 구문

기본 구조

let i = 0; // 초기값

while (i < 5) { // 실행 조건
  console.log(`현재 i의 값: ${i}`);
  i++; // 갱신
}

console.log('루프가 종료되었습니다.');

초기값

반복을 시작하기 전 설정

조건

매번 확인하는 반복 조건

갱신

조건 변경을 위한 값 수정

for 구문

구조 분해

for (let i = 0; i < 5; i++) {
     ↑초기값   ↑조건  ↑갱신
  console.log(`현재 i의 값: ${i}`);
}

while vs for 비교

구분 while for
사용 시기 조건 중심 반복 횟수 정해진 반복
초기값 별도 선언 괄호 내 선언
갱신 블록 내부 괄호 내 명시

배열 순회

전통적인 for

const arr = [1, 2, 3, 4, 5];
for (let i = 0; i < arr.length; i++) {
  console.log(arr[i]);
}

forEach 메소드

const arr = [1, 2, 3, 4, 5];
arr.forEach((item, index) => {
  console.log(item);
});

for...of 구문

const arr = [1, 2, 3, 4, 5];
for (let item of arr) {
  console.log(item);
}

break & continue

break

루프를 완전히 종료 
// 퀴즈 예제
while (true) {
  const answer = prompt('빨강의 보색은?');
  if (answer === '초록') {
    alert('정답입니다! 🎉');
    break; // 루프 종료
  } else {
    alert('틀렸습니다! 다시 시도해보세요.');
  }
}

continue

현재 반복을 건너뛰고 다음으로 
// 7의 배수만 출력
for (let i = 1; i < 100; i++) {
  if (i % 7 !== 0) {
    continue; // 나머지 코드 건너뛰기
  }
  console.log(`${i}는 7의 배수입니다.`);
}
