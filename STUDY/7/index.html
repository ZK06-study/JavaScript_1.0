왜 "없음"을 나타내는 값이 두 개일까?

개발자의 고민

let name;        // 아직 정하지 않음
let age = null;  // 의도적으로 비워둠

// 둘 다 "없음"인데... 무슨 차이가 있지?

같은 "없음"인데 표현 방법이 다른 이유는?

undefined: "아직 정해지지 않음"

JavaScript가 자동으로 부여하는 값

개발자가 명시적으로 할당하지 않은 상태

선언만 된 변수

let name;
console.log(name); // undefined

없는 객체 속성

const obj = {};
console.log(obj.prop); // undefined

반환값 없는 함수

function hello() {
  console.log('hi');
}
console.log(hello()); // undefined

null: "의도적으로 비워둠"

개발자가 명시적으로 할당하는 값

"여기에는 의도적으로 아무것도 없다"는 의미

의도적 초기화

let currentUser = null;
// "현재 로그인한 사용자 없음"

데이터베이스 NULL

const user = {
  name: '김철수',
  middleName: null  // 중간 이름 없음
};

초기화/리셋

let selectedItem = getItem();
// 선택 해제
selectedItem = null;

핵심 차이점 비교

undefined

할당 주체: JavaScript 엔진
의미: 아직 정해지지 않음
타입: 'undefined'
권장 사용: 직접 할당 금지
VS

null

할당 주체: 개발자
의미: 의도적으로 비어있음
타입: 'object' (버그)
권장 사용: 명시적 "없음" 표현

null

JavaScript의 유명한 버그!

왜 null의 typeof가 'object'일까?

JavaScript 초기 설계의 실수로, 하위 호환성을 위해 수정되지 않았습니다.

// 올바른 null 체크 방법
value === null  // ✅ 정확
typeof value === 'object' && value === null  // ❌ 복잡

동등성 비교: == vs ===

느슨한 비교 (==)

null == undefined true
null == 0 false
null == '' false
null == false false

엄격한 비교 (===)

null === undefined false
null === null true
undefined === undefined true

핵심 포인트

== 연산자는 null과 undefined를 서로 같다고 판단하지만, ===는 다르다고 판단합니다.

안전한 null check 방법

권장 방법 1: == 활용

if (value != null) {
  // value가 null도 undefined도 아닐 때
  console.log('값이 존재합니다:', value);
}

null과 undefined를 한 번에 체크

엄격한 방법: === 활용

if (value !== null && value !== undefined) {
  console.log('값이 존재합니다:', value);
}

명확하지만 길어짐

현대적 방법: nullish coalescing

const result = value ?? '기본값';
// value가 null이나 undefined면 '기본값' 사용

ES2020의 새로운 연산자

실무 권장사항

권장하는 방법

  • 명시적 "없음" 표현 시 null 사용
  • null check는 != null 사용
  • undefined를 직접 할당하지 않기
  • 객체 속성의 부재는 아예 정의하지 않기
// ✅ 좋은 예
let currentUser = null;
const user = { name: 'Kim' }; // address 속성 없음

if (currentUser != null) {
  showProfile(currentUser);}

피해야 할 방법

  • undefined를 직접 할당하기
  • null 체크 시 typeof 사용하기
  • 불필요한 속성에 null 할당하기
// ❌ 나쁜 예
let name = undefined; // 의미 모호
const user = { 
  name: 'Kim', 
  address: undefined // 불필요
};

if (typeof value === 'object' && value === null) {
  // 복잡한 null 체크
}

Before & After: 코드 개선하기

Before (개선 전)

// 혼란스러운 코드
let userName = undefined;
let userAge = undefined;
let userEmail = undefined;

const userData = {
  name: userName,
  age: userAge, 
  email: userEmail,
  phone: undefined
};

// 복잡한 체크
if (typeof userData.name !== 'undefined' && 
    userData.name !== null) {
  displayName(userData.name);
}

After (개선 후)

// 명확한 코드
let userName = null;
let userAge = null;  
let userEmail = null;

const userData = {
  name: userName,
  age: userAge,
  email: userEmail
  // phone 속성은 아예 제외
};

// 간단한 체크
if (userData.name != null) {
  displayName(userData.name);
}

개선된 점

  • 의도가 명확해짐 (null = 의도적 빈 값)
  • 불필요한 속성 제거
  • 간단한 null check 사용

console.log(typeof b);     // ?

💡 정답:

'undefined'  // typeof a
'object'     // typeof b
true         // a == b
false        // a === b

문제 2: 안전한 함수 만들기

사용자 정보를 안전하게 출력하는 함수를 완성하세요.

function printUserInfo(user) {
  // user가 null/undefined이면 "정보 없음" 출력
  // 아니면 user.name을 출력하는 코드 작성
}

printUserInfo(null);           // "정보 없음"
printUserInfo({name: '김철수'}); // "김철수"

💡 정답:

function printUserInfo(user) {
  if (user != null) {
    console.log(user.name);
  } else {
    console.log("정보 없음");
  }
}
