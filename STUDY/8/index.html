매개변수와 인수

매개변수 (Parameter)

함수 정의 시 선언하는 변수

function add(x, y) {
//           ↑   ↑
//      매개변수들
}

인수 (Argument)

함수 호출 시 전달하는 실제 값

add(2, 3);
//  ↑  ↑
// 인수들

매개변수의 독립성

중요한 특징

매개변수는 바깥 변수와 독립적입니다. 함수 내부에서 매개변수 값을 변경해도 원본 변수는 영향받지 않습니다.

실제 예제

function reassign(x) {
  x = 100;  // 매개변수 x 변경
  return x;
}

const y = 50;
const result = reassign(y);

console.log(y);      // 50 (변경되지 않음)
console.log(result); // 100

반환값 (Return Value)

return 구문

함수의 실행 결과를 호출한 곳으로 돌려보내는 구문

동작 방식

  • return 키워드 다음의 값이 반환됨
  • 반환되는 즉시 함수 실행이 종료됨
  • return 없으면 undefined 반환

값 반환

function add(x, y) {
  return x + y;
}
add(2, 3); // 5

조기 종료

function add(x, y) {
  return x + y;
  console.log('실행 안됨');
}

undefined 반환

function noReturn() {
  // return 없음
}
noReturn(); // undefined

스코프 (Scope)

스코프란?

변수가 유효한 코드 상의 범위를 의미합니다.

기본 예제

function add(x, y) { // x, y는 함수 내부에서만 유효
  return x + y;
}
add(2, 3);
console.log(x); // 에러! x는 함수 밖에서 접근 불가

스코프의 장점

변수 보호

함수 내부 변수가 외부에 노출되지 않아 안전

이름 재사용

같은 변수명을 다른 함수에서 사용 가능

메모리 효율

함수 종료 시 지역 변수가 자동으로 정리

스코프 연쇄 (Scope Chain)

스코프 연쇄란?

변수를 찾을 때 현재 스코프부터 시작해서 바깥쪽 스코프로 차례대로 찾아가는 과정

인터랙티브 데모

const five = 5;          // 전역 스코프

function add5(x) {        // 함수 스코프
  function add(y) {       // 중첩 함수 스코프
    return x + y;         // x는 상위 스코프에서 찾음
  }
  return add(five);       // five는 전역 스코프에서 찾음
}

add5(3); // 8

변수 찾기 과정

1

현재 스코프에서 변수 검색

2

없으면 상위 스코프로 이동

3

전역 스코프까지 반복

4

끝까지 없으면 에러 발생

변수 가리기 (Variable Shadowing)

변수 가리기란?

안쪽 스코프에서 바깥쪽 스코프와 같은 이름의 변수를 선언하면, 안쪽에서는 바깥쪽 변수가 가려지는 현상

인터랙티브 데모

const x = 3;              // 전역 x

function add5(x) {        // 매개변수 x (전역 x를 가림)
  function add(x, y) {    // 매개변수 x (상위 x를 가림)
    return x + y;         // 가장 가까운 x 사용
  }
  return add(x, 5);       // 함수 스코프의 x 사용
}

add5(10); // 15

주의사항

  • 가려진 변수는 접근 불가능해짐
  • 같은 스코프에서는 같은 이름 변수 선언 불가
  • 의도치 않은 가리기는 버그의 원인이 될 수 있음

어휘적 스코핑 (Lexical Scoping)

어휘적 스코핑이란?

스코프가 코드가 작성된 구조에 의해 결정되며, 함수 호출 위치와는 무관한 특성

잘못된 이해

function add5(x) {
  const five = 5;
  return add(x);    // add가 여기서 호출됨
}

function add(x) {
  return five + x;  // ❌ five에 접근 불가
}

add5(3); // ReferenceError

호출 위치와 상관없이 add 함수는 five 변수에 접근할 수 없습니다.

올바른 방법

function add5(x) {
  const five = 5;
  
  function add(x) {   // add를 내부에 정의
    return five + x;  // ✅ five에 접근 가능
  }
  
  return add(x);
}

add5(3); // 8

함수가 정의된 위치에 따라 스코프가 결정됩니다.

값으로서의 함수

1급 함수 (First-Class Function)

JavaScript에서 함수는 값으로 취급되어 다른 데이터 타입처럼 자유롭게 사용할 수 있습니다.

1급 함수의 특징

변수 할당

const myFunc = function() {
  return "Hello!";
};

인수로 전달

[1,2,3].filter(function(x) {
  return x > 1;
});

반환값으로 사용

function createFunc() {
  return function() {
    return "Created!";
  };
}

자료구조에 저장

const funcs = [add, subtract];
const obj = {method: myFunc};

화살표 함수 (Arrow Function)

화살표 함수란?

ES2015에서 도입된 간결한 함수 정의 문법으로, 특히 익명 함수를 간단하게 작성할 때 유용합니다.

문법 비교

전통적인 함수

// 기본 형태
function add(x, y) {
  return x + y;
}

// 익명 함수
const add = function(x, y) {
  return x + y;
};

화살표 함수

// 기본 형태
const add = (x, y) => {
  return x + y;
};

// 간단한 형태
const add = (x, y) => x + y;

다양한 형태

매개변수 하나

const negate = x => !x;
// 괄호 생략 가능

매개변수 없음

const sayHello = () => "Hello!";
// 빈 괄호 필요

블록 구문

const complex = (x, y) => {
  const result = x * y;
  return result + 1;
};

객체 반환

const createUser = name => ({
  name: name,
  id: Math.random()
});
